name: Yologuard CI
on:
  push:
    paths:
      - 'code/yologuard/**'
      - 'tests/**'
  pull_request:
    paths:
      - 'code/yologuard/**'
      - 'tests/**'

jobs:
  pytest:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.12'
      - name: Install requirements
        run: |
          python -m pip install --upgrade pip
          pip install pytest
      - name: Run Yologuard unit tests
        run: |
          pytest tests/test_yologuard.py

  yologuard-scan:
    runs-on: ubuntu-latest
    needs: pytest
    steps:
      - uses: actions/checkout@v4
      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.12'
      - name: Run IF.yologuard_v3 in dry-run mode
        run: |
          python code/yologuard/src/IF.yologuard_v3.py docs infra --dry-run
        env:
          YG_SKIP: 'tests,out'
